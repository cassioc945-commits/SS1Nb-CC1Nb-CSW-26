            import math

            def resolver_segundo_grau(a, b, c):
                if a == 0:
                    return "Não é equação do 2º grau (a = 0)."
    
                delta = b**2 - 4*a*c
    
                if delta < 0:
                    return "Não existem raízes reais."
                elif delta == 0:
                    x = -b / (2*a)
                    return f"Uma raiz real: x = {x}"
                else:
                    x1 = (-b + math.sqrt(delta)) / (2*a)
                    x2 = (-b - math.sqrt(delta)) / (2*a)
                    return f"Duas raízes reais: x1 = {x1}, x2 = {x2}"

                # Exemplo de uso:
                a = 1
                b = -3
                c = 2

                resultado = resolver_segundo_grau(a, b, c)
